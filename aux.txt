CREATE TABLE users (id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, login TEXT, name TEXT, bio TEXT);

CREATE TABLE follows (
	follower INT, 
	followed INT, 
	PRIMARY KEY(follower, followed),
	FOREIGN KEY (follower) REFERENCES users (id) ON DELETE CASCADE,
	FOREIGN KEY (followed) REFERENCES users (id) ON DELETE CASCADE );S
